# ============================================================
# Zealasm Long Mnemonic Test
# Tests that an extremely long invalid mnemonic doesn't overflow
# the directive buffer and corrupt assembler state.
# The assembler should report an error for the bad line but
# not crash. Then assemble a clean file to prove state is OK.
# Expected: first assemble has error, second succeeds, prints "OK"
# ============================================================
# --- Build BAD.ASM with a very long mnemonic ---
A:ED
i
    org 0
    abcdefghijklmnopqrstuvwxyz 1234
    halt
.
w BAD.ASM
Q
# --- Build GOOD.ASM (clean) ---
A:ED
i
    org 0
    jp main
cono equ 0x82
cwstr equ 16
sysexit equ 0
kernel equ 0x0010
main:
    ld de, msg
    ld b, cono
    ld c, cwstr
    call kernel
    ld c, sysexit
    call kernel
msg:
    defm "OK"
    defb 0x0D
    defb 0x0A
    defb 0
.
w GOOD.ASM
Q
# --- Assemble bad file (expect error, should not crash) ---
ZEALASM.APP BAD.ASM BAD.APP
# --- Assemble and run good file (proves assembler state is intact) ---
ZEALASM.APP GOOD.ASM GOOD.APP
GOOD.APP
HALT
